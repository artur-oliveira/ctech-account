# KYC no Account — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Identity verification (CPF + legal name + birth date) with levels `none → basic → verified`, a `kyc_level` JWT claim behind a new `kyc` scope, and internal endpoints (via a new `client_credentials` grant) so ctech-wallet can confirm verification on the first PIX deposit.

**Architecture:** New `internal/domain/kyc` package (service + repository over the existing `account_users` table, with a `CPF_{cpf}` uniqueness item written transactionally). New `client_credentials` grant restricted to confidential first-party clients. `kyc_level` claim added to access token / id_token / userinfo when the `kyc` scope is granted. UI gets an `/account/identity` page.

**Tech Stack:** Go 1.26, Fiber v3, DynamoDB (TransactWriteItems), Next.js 16 / React 19 / ShadCN, react-query.

**Spec:** `docs/specs/2026-07-10-kyc-design.md`

## Global Constraints

- All errors via `*apierror.Problem` + `problem.Send(c)`; validation via `validate.Struct(req)`.
- Constants for every literal (scope names, event types, levels, grant types).
- `c.Context()` — never `c.UserContext()`.
- Layering handler → service → repository (interfaces only in services).
- CPF never appears in tokens, audit metadata, or logs. Masked as `***.***.***-XX` on user-facing responses.
- Tests: unit in `internal/domain/kyc/service_test.go`, integration in `internal/handler/kyc_test.go`; mocks in `testhelpers_test.go`.
- Build/test commands: `go build ./cmd/... ./internal/...` and `go test ./internal/...` (never bare `./...` — cdk/node_modules breaks it). `gofmt -w` touched files before each commit.
- Commit per task, Conventional Commits, no emojis.

```go
package kyc

// IsValidCPF reports whether cpf is 11 numeric digits with valid check digits.
// Sequences of a single repeated digit (e.g. 11111111111) pass the checksum
// but are not real CPFs, so they are rejected explicitly.
func IsValidCPF(cpf string) bool {
	if len(cpf) != 11 {
		return false
	}
	allSame := true
	for i := 0; i < 11; i++ {
		if cpf[i] < '0' || cpf[i] > '9' {
			return false
		}
		if cpf[i] != cpf[0] {
			allSame = false
		}
	}
	if allSame {
		return false
	}
	return checkDigit(cpf, 9) == int(cpf[9]-'0') && checkDigit(cpf, 10) == int(cpf[10]-'0')
}

// checkDigit computes the CPF verification digit at position pos (9 or 10)
// from the preceding pos digits, per the Receita Federal mod-11 algorithm.
func checkDigit(cpf string, pos int) int {
	sum := 0
	for i := 0; i < pos; i++ {
		sum += int(cpf[i]-'0') * (pos + 1 - i)
	}
	d := 11 - sum%11
	if d >= 10 {
		return 0
	}
	return d
}

// MaskCPF renders a CPF as ***.***.***-XX (last two digits visible).
func MaskCPF(cpf string) string {
	if len(cpf) != 11 {
		return ""
	}
	return "***.***.***-" + cpf[9:]
}
```

```go
package kyc

import "errors"

// KYC verification levels stored on the user and exposed as the kyc_level claim.
const (
	LevelNone     = ""
	LevelBasic    = "basic"    // CPF checksum + age validated, awaiting PIX match
	LevelVerified = "verified" // first PIX deposit matched the declared CPF
)

// MinAge is the minimum age (years) to reach LevelBasic — real-money games
// require adults, and the rule lives here so downstream services only read
// kyc_level.
const MinAge = 18

// BuildCPFPK keys the uniqueness item enforcing one CPF per account.
func BuildCPFPK(cpf string) string {
	return "CPF_" + cpf
}

var (
	ErrInvalidCPF      = errors.New("invalid cpf")
	ErrUnderage        = errors.New("user is under the minimum age")
	ErrCPFConflict     = errors.New("cpf already registered to another account")
	ErrAlreadyVerified = errors.New("kyc already verified")
	ErrCPFMismatch     = errors.New("cpf does not match the declared one")
	ErrNotSubmitted    = errors.New("kyc data not submitted")
)

// Status is the user-facing view of KYC state (CPF always masked).
type Status struct {
	Level      string `json:"level"`
	CPFMasked  string `json:"cpf_masked,omitempty"`
	LegalName  string `json:"legal_name,omitempty"`
	BirthDate  string `json:"birth_date,omitempty"`
	VerifiedAt string `json:"verified_at,omitempty"`
}
```

```go
package kyc

import (
	"context"
	"errors"
	"fmt"
	"time"

	"gopkg.aoctech.app/account/internal/database"
	"gopkg.aoctech.app/account/internal/domain/user"
	"github.com/aws/aws-sdk-go-v2/aws"
	"github.com/aws/aws-sdk-go-v2/feature/dynamodb/attributevalue"
	"github.com/aws/aws-sdk-go-v2/service/dynamodb/types"
)

// usersTable matches the table used by the user repository — the CPF
// uniqueness item lives next to the user items (single-table pattern).
const usersTable = "account_users"

const conditionalCheckFailed = "ConditionalCheckFailed"

// Repository persists KYC state on the user item plus a CPF_{cpf}
// uniqueness item, transactionally.
type Repository interface {
	GetUser(ctx context.Context, userID string) (*user.User, error)
	// SaveSubmission writes cpf/legal_name/birth_date and kyc_level=basic on the
	// user, claims CPF_{cpf} (failing with ErrCPFConflict if another account owns
	// it), and releases CPF_{oldCPF} when re-submitting with a different CPF.
	SaveSubmission(ctx context.Context, userID, cpf, legalName, birthDate, oldCPF string) error
	MarkVerified(ctx context.Context, userID, verifiedAt string) error
}

type dynamoRepository struct {
	db       *database.Client
	userRepo user.Repository
}

// NewRepository returns a DynamoDB-backed Repository reusing the user
// repository for reads.
func NewRepository(db *database.Client, userRepo user.Repository) Repository {
	return &dynamoRepository{db: db, userRepo: userRepo}
}

func (r *dynamoRepository) GetUser(ctx context.Context, userID string) (*user.User, error) {
	return r.userRepo.GetByID(ctx, userID)
}

func (r *dynamoRepository) SaveSubmission(ctx context.Context, userID, cpf, legalName, birthDate, oldCPF string) error {
	table := r.db.TableName(usersTable)
	now := time.Now().UTC().Format(time.RFC3339)

	cpfItem, err := attributevalue.MarshalMap(map[string]string{
		"pk":         BuildCPFPK(cpf),
		"user_id":    userID,
		"created_at": now,
	})
	if err != nil {
		return fmt.Errorf("marshaling cpf item: %w", err)
	}

	items := []types.TransactWriteItem{
		{
			Put: &types.Put{
				TableName:           aws.String(table),
				Item:                cpfItem,
				ConditionExpression: aws.String("attribute_not_exists(pk)"),
			},
		},
		{
			Update: &types.Update{
				TableName: aws.String(table),
				Key: map[string]types.AttributeValue{
					"pk": &types.AttributeValueMemberS{Value: user.BuildPK(userID)},
				},
				UpdateExpression: aws.String(
					"SET cpf = :cpf, legal_name = :ln, birth_date = :bd, kyc_level = :lvl, updated_at = :now",
				),
				ExpressionAttributeValues: map[string]types.AttributeValue{
					":cpf": &types.AttributeValueMemberS{Value: cpf},
					":ln":  &types.AttributeValueMemberS{Value: legalName},
					":bd":  &types.AttributeValueMemberS{Value: birthDate},
					":lvl": &types.AttributeValueMemberS{Value: LevelBasic},
					":now": &types.AttributeValueMemberS{Value: now},
				},
			},
		},
	}

	if oldCPF != "" && oldCPF != cpf {
		items = append(items, types.TransactWriteItem{
			Delete: &types.Delete{
				TableName: aws.String(table),
				Key: map[string]types.AttributeValue{

			},
		})

		if errors.As(err, &canceled) {
			for _, reason := range canceled.CancellationReasons {
				if reason.Code != nil && *reason.Code == conditionalCheckFailed {
					return ErrCPFConflict
				}
			}
		}
		return err
	}
	return nil
}
```

```go
package kyc

import (
	"context"
	"fmt"
	"time"

const birthDateLayout = "2006-01-02"

// Service implements the KYC state machine: none → basic (Submit) →
// verified (Confirm, called by trusted services after a PIX CPF match).
type Service struct {
	repo Repository
}

func NewService(repo Repository) *Service {
	return &Service{repo: repo}
}

// Submit validates CPF checksum and minimum age, then persists the
// submission at LevelBasic. Rejected entirely (nothing persisted) on any
// validation failure. Re-submission is allowed until the user is verified.
func (s *Service) Submit(ctx context.Context, userID, cpf, legalName, birthDate string) error {
	if !IsValidCPF(cpf) {
		return ErrInvalidCPF
	}
	born, err := time.Parse(birthDateLayout, birthDate)
	if err != nil {
		return fmt.Errorf("%w: bad birth date", ErrInvalidCPF)
	}
	if !isAtLeast(born, MinAge) {
		return ErrUnderage
	}

	u, err := s.repo.GetUser(ctx, userID)
	if err != nil {
		return err
	}
	if u.KYCLevel == LevelVerified {
		return ErrAlreadyVerified
	}

	if u.CPF == cpf {
		// Same CPF, updated name/birth date: the uniqueness item already belongs
		// to this user — a conditional Put would spuriously conflict.
		return s.repo.SaveSubmission(ctx, userID, cpf, legalName, birthDate, cpf)
	}
	return s.repo.SaveSubmission(ctx, userID, cpf, legalName, birthDate, u.CPF)
}

// Confirm marks the user verified after the caller (wallet) matched a PIX
// deposit's payer CPF against the declared one. Idempotent when already
// verified with the same CPF.
func (s *Service) Confirm(ctx context.Context, userID, cpf string) error {
	u, err := s.repo.GetUser(ctx, userID)
	if err != nil {
		return err
	}
	if u.CPF == "" || u.KYCLevel == LevelNone {
		return ErrNotSubmitted
	}
	if u.CPF != cpf {
		return ErrCPFMismatch
	}
	if u.KYCLevel == LevelVerified {
		return nil
	}
	return s.repo.MarkVerified(ctx, userID, time.Now().UTC().Format(time.RFC3339))
}

// Get returns the user-facing KYC status (CPF masked).
func (s *Service) Get(ctx context.Context, userID string) (*Status, error) {
	u, err := s.repo.GetUser(ctx, userID)
	if err != nil {
		return nil, err
	}
	return &Status{
		Level:      u.KYCLevel,
		CPFMasked:  MaskCPF(u.CPF),
		LegalName:  u.LegalName,
		BirthDate:  u.BirthDate,
		VerifiedAt: u.KYCVerifiedAt,
	}, nil
}

// isAtLeast reports whether someone born on born is at least years old today
// (UTC). The comparison uses date arithmetic so the birthday itself counts.
func isAtLeast(born time.Time, years int) bool {
	return !time.Now().UTC().Before(born.AddDate(years, 0, 0))
}
```

```go
// clientCredentials issues a machine-to-machine token (RFC 6749 §4.4).
// Restricted to confidential first-party clients: internal:* scopes gate
// service-to-service APIs, and third parties must never reach them.
func (h *TokenHandler) clientCredentials(c fiber.Ctx) error {
	clientID := c.FormValue("client_id")
	clientSecret := c.FormValue("client_secret")
	if clientID == "" || clientSecret == "" {
		return apierror.InvalidRequest("client_id and client_secret are required.", c.Path()).Send(c)
	}

	oauthClient, err := h.clientRepo.GetByID(c.Context(), clientID)
	if err != nil {
		if errors.Is(err, oauthclient.ErrNotFound) {
			return apierror.InvalidClient("Unknown client_id.", c.Path()).Send(c)
		}
		return apierror.ServerError(c.Path()).Send(c)
	}

	if oauthClient.IsPublic() || !oauthClient.FirstParty {
		return apierror.UnauthorizedClient(c.Path()).Send(c)
	}
	ok, _ := crypto.VerifyPassword(clientSecret, oauthClient.ClientSecretHash)
	if !ok {
		return apierror.InvalidClient("Invalid client_secret.", c.Path()).Send(c)
	}

	requested := strings.Fields(c.FormValue("scope"))
	scp := oauthClient.FilterScopes(requested)
	if len(scp) == 0 {
		return apierror.InvalidScope(c.Path()).Send(c)
	}

	serviceAudiences, err := h.catalogSvc.AudiencesFor(c.Context(), scp)
	if err != nil {
		return apierror.ServerError(c.Path()).Send(c)
	}
	audience := append([]string{h.cfg.Audience}, serviceAudiences...)

	// sub = client_id, no session: internal middleware relies on the empty sid
	// to tell machine tokens from user tokens. No step-up claims, no kyc_level.
	accessToken, err := h.jwtSvc.SignAccessToken(clientID, "", clientID, scp, h.baseURL, audience, 0, 0, nil, "")
	if err != nil {
		return apierror.ServerError(c.Path()).Send(c)
	}

	return c.JSON(fiber.Map{
		"access_token": accessToken,
		"token_type":   "Bearer",
		"expires_in":   accessTokenTTLSeconds,
		"scope":        strings.Join(scp, " "),
	})
}
```

```go
	}
	if err := validate.Struct(req); err != nil {
		if ve, ok := validate.IsValidationError(err); ok {
			return apierror.ValidationFailed(ve.Detail(), c.Path()).Send(c)
		}
		return apierror.ServerError(c.Path()).Send(c)
	}

	if err := h.kycSvc.Submit(c.Context(), userID, req.CPF, req.LegalName, req.BirthDate); err != nil {
		switch {
		case errors.Is(err, kyc.ErrInvalidCPF):
			return apierror.ValidationFailed("cpf: invalid CPF.", c.Path()).Send(c)
		case errors.Is(err, kyc.ErrUnderage):
			return apierror.AgeRequirementNotMet(c.Path()).Send(c)
		case errors.Is(err, kyc.ErrAlreadyVerified):
			return apierror.KYCAlreadyVerified(c.Path()).Send(c)
		case errors.Is(err, kyc.ErrCPFConflict):
			return apierror.CPFAlreadyRegistered(c.Path()).Send(c)
		case errors.Is(err, user.ErrNotFound):
			return apierror.NotFound("User", c.Path()).Send(c)
		}
		return apierror.ServerError(c.Path()).Send(c)
	}

	recordAudit(c, h.audit, userID, audit.EventKYCSubmitted, nil)

	st, err := h.kycSvc.Get(c.Context(), userID)
	if err != nil {
		return apierror.ServerError(c.Path()).Send(c)
	}
	return c.JSON(st)
}

func (h *KYCHandler) get(c fiber.Ctx) error {
	st, err := h.kycSvc.Get(c.Context(), middleware.GetUserID(c))
	if err != nil {
		if errors.Is(err, user.ErrNotFound) {
			return apierror.NotFound("User", c.Path()).Send(c)
		}
		return apierror.ServerError(c.Path()).Send(c)
	}
	return c.JSON(st)
}

type confirmKYCRequest struct {
	UserID string `json:"user_id" validate:"required"`
	CPF    string `json:"cpf" validate:"required,len=11,numeric"`
}

func (h *KYCHandler) confirm(c fiber.Ctx) error {
	var req confirmKYCRequest
	if err := c.Bind().Body(&req); err != nil {
		return apierror.InvalidRequest("Invalid request body.", c.Path()).Send(c)
	}
	if err := validate.Struct(req); err != nil {
		if ve, ok := validate.IsValidationError(err); ok {
			return apierror.ValidationFailed(ve.Detail(), c.Path()).Send(c)
		}
		return apierror.ServerError(c.Path()).Send(c)
	}

	// callerClientID: for client_credentials tokens sub == client_id.
	callerClientID := middleware.GetUserID(c)

	if err := h.kycSvc.Confirm(c.Context(), req.UserID, req.CPF); err != nil {
		switch {
		case errors.Is(err, kyc.ErrCPFMismatch):
			recordAudit(c, h.audit, req.UserID, audit.EventKYCConfirmFailed, map[string]string{"client_id": callerClientID})
			return apierror.KYCCPFMismatch(c.Path()).Send(c)
		case errors.Is(err, kyc.ErrNotSubmitted):
			return apierror.KYCNotSubmitted(c.Path()).Send(c)
		case errors.Is(err, user.ErrNotFound):
			return apierror.NotFound("User", c.Path()).Send(c)
		}
		return apierror.ServerError(c.Path()).Send(c)
	}

	recordAudit(c, h.audit, req.UserID, audit.EventKYCVerified, map[string]string{"client_id": callerClientID})
	return c.JSON(fiber.Map{"confirmed": true})
}

// internalGet returns the full (unmasked) identity record — service-to-service

	})
}
```
